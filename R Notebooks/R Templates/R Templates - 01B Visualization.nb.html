Andrews Curves

See the package vignette with `vignette("andrews")`

data(iris)

Barplot

library(mlbench)
# load the dataset
data(BreastCancer)
# create a bar plot of each categorical attribute
par(mfrow=c(2,4))
for(i in 2:9) {
    counts <- table(BreastCancer[,i])
    name <- names(BreastCancer)[i]
    barplot(counts, main=name)
}

Boxplot by Class

library(caret)
# load the iris dataset
data(iris)
# box and whisker plots for each attribute by class value
x <- iris[,1:4]
y <- iris[,5]
featurePlot(x=x, y=y, plot="box")

Correlation Plot

library(corrplot)
corrplot 0.92 loaded
# load the data
data(iris)
# calculate correlations
correlations <- cor(iris[,1:4])
# create correlation plot
corrplot(correlations, method="circle")

Density Plot

library(lattice)
# load dataset
data(iris)
# create a panel of simpler density plots by attribute
par(mfrow=c(1,4))
for(i in 1:4) {
    plot(density(iris[,i]), main=names(iris)[i])
}

Density Plot by Class

library(caret)
# load the data
data(iris)
# density plots for each attribute by class value
x <- iris[,1:4]
y <- iris[,5]
scales <- list(x=list(relation="free"), y=list(relation="free"))
featurePlot(x=x, y=y, plot="density", scales=scales)

Histogram
